// n1-натуральное, кол-во цифр n1.n=m (n1 = a[m] a[m-1] .. a[2] a[1])
// n2-натуральное, кол-во цифр n2.n=k (n2 = b[k] b[k-1] .. b[2] b[1])


ФУНКЦИЯ COM_NN_D(НАТУРАЛЬНОЕ n1, НАТУРАЛЬНОЕ n2)
	m := n1.n
	k := n2.n
	ЕСЛИ (m > k) ТО
		ВОЗВРАТ 2
	ИНАЧЕ ЕСЛИ (k > m) ТО
		ВОЗВРАТ 1
	ИНАЧЕ
		i := m
		ЦИКЛ ПОКА ((a[i] = b[i]) И (i >= 0))
			i:=i-1
		КЦ
		ЕСЛИ (i < 0) ТО
			ВОЗВРАТ 0
		ИНАЧЕ ЕСЛИ (a[i] > b[i]) ТО
			ВОЗВРАТ 2
		ИНАЧЕ 
			ВОЗВРАТ 1
		КЕ
	КЕ
КФ
