/* Беляева
c1 – делимое, c2 – делитель, 
m1, m2 – их степени соответственно
Ссылается на - DIV_PP_P, MUL_PP_P, SUB_PP_P
*/

ФУНКЦИЯ MOD_PP_P(многочлен c1, многочлен c2)
	ЕСЛИ p2.m = 0 И p2[0] = 0 ТО
		Ошибка
	ИНАЧЕ
		ЕСЛИ (с1.m < c2.m), ТО
			r := c1
		ИНАЧЕ
			q := DIV_PP_P(c1, c2)
			c3 := MUL_PP_P(c2, q)
			r := SUB_PP_P(c1, c3)
		КЕ
		ВЕРНУТЬ r
	КЕ
КФ
