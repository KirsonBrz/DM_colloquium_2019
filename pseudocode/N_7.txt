//Умножение натурального числа на 10^k

ФУНКЦИЯ MUL_Nk_N (натуральное N1, степень k)
	ЕСЛИ (N1.n != 0 или N1[0] != 0) то
		N := N1.копия
		заполнить N k нулями
		N.n := N.n + k
		ЦИКЛ ПО i ОТ N.n - k ДО 0
			N[i+k] := N[i]
		КЦ
		ЦИКЛ ПО i ОТ 0 ДО k
			N[i] := 0
		КЦ
	ИНАЧЕ
		N = натуральное()
	КЕ
	Вернуть N
КФ
