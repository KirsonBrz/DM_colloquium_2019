/* Абдуллин
/ Преобразование многочлена — кратные корни в простые
f-начальный массив коэффициентов
p-массив коэффициентов производной f
Q-массив коэффициентов НОД(f,p)
Ссылается на - DER_P_P, GCF_PP_P, DIV_PP_P
*/

ФУНКЦИЯ NMR_P_P (многочлен f)
	Q := многочлен(1, 1, 1)
	ПОКА Q != 0
		p := DER_P_P(f)
		Q := GCF_PP_P(f, p)
		ЕСЛИ Q != 0
			f:=DIV_PP_P(f, Q)
	КЦ
	ВОЗВРАТ f
КФ
