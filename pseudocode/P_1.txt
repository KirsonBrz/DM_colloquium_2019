/*
c1, c2 – заданные многочлены, 
с1.m, c2.m – их степени соответственно, 
с – полученный многочлен со степенью m
*/

// Ссылается на - ADD_QQ_Q

ФУНКЦИЯ ADD_PP_P (многочлен c1, многочлен c2)
	m1 := c1.m
	m2 := c2.m
	c := создать_многочлен ()
	
	ECЛИ (m1 > m2), TO 
		ПОМЕНЯТЬ m1 и m2
		ПОМЕНЯТЬ c1 И c2
	KE
	
	i := 0
	ЦИКЛ ПОКА (i <= m2)
		ЕСЛИ (i <= m1), ТО 
			c[i] := ADD_QQ_Q(c1[i], c2[i])
		ИНАЧЕ 
			c[i] := c2[i]
		КЕ
		i := i+1
	КЦ
	c.m := m2
	ВЕРНУТЬ c
КФ
