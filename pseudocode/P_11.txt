// Волконенков
// Ссылается на - MOD_PP_P, DEG_P_N

функция GCF_PP_P(многочлены A, B):
	C = A.копия
	D = B.копия
	nod = многочлен 
	ЦИКЛ ПОКА (C.m !=0 ИЛИ D.m!=0): 
		ЕСЛИ (C.m > D.m): 
			C = MOD_PP_P(C,D) 
			C.m = DEG_P_N(C) 
		ИНАЧЕ: 
			D := MOD_PP_P(D,C) 
			D.m := DEG_P_N(D) 
		КЕ
	КЦ
	ЕСЛИ (D.m != 0): 
		ЦИКЛ ПО i от 0 до C.m: 
			nod[i] := C[i] 
		КЦ
	ИНАЧЕ: 
		ЦИКЛ ПО i от 0 до D.m:
			nod[i] := D[i]
		КЦ
	ке
	Вернуть nod
кф
