функция GCF_PP_P(многочлены A, B):
	a = A.копия
	b = B.копия
	nod = многочлен 
	ЦИКЛ ПОКА (a.m !=0 ИЛИ b.m!=0): 
		ЕСЛИ (a.m > b.m)
			a = MOD_PP_P(a,b) 
			a.m = DEG_P_N(a) 
		ИНАЧЕ 
			b := MOD_PP_P(b, a) 
			b.m := DEG_P_N(b) 
		КЕ
	КЦ
	ЕСЛИ (b.m != 0): 
		nod := a.копия
	ИНАЧЕ
		nod := b.копия
	ке
	nod.m := DEG_P_N(nod)
	
	Вернуть nod
