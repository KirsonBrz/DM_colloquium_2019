ФУНКЦИЯ ADD_NN_N (натуральные A, B)
	// Если количество разрядов второго числа больше первого 
	ЕСЛИ (A.n < B.n) 
		Поменять A и B
	КОНЕЦ ЕСЛИ
	
	C := натуральное
	Очистить C
	k := 0
	i := 0
	// Сложение
	ЦИКЛ ПОКА (i <= A.n)
		Если i <= B.n
			C.добавить_в_конец(B[i] + A[i] +k)
		ИНАЧЕ
			C.добавить_в_конец(A[i] + k)
		ЕСЛИ (C[i] >= 10) 
			C[i] := C[i] - 10;
			k := 1
		ИНАЧЕ
			k := 0
		КОНЕЦ ЕСЛИ
		i := i+1
	КОНЕЦ ЦИКЛА
	// Если количество разрядов у суммы больше чем у исходных 
	C.n := A.n;
	ЕСЛИ (k = 1)
		C.добавить_в_конец(1)
		C.n := C.n + 1
	КОНЕЦ ЕСЛИ
	
	ВЕРНУТЬ С 
КОНЕЦ ФУНКЦИИ
