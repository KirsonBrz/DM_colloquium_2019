MUL_ND_N,COM_NN_D, SUB_NN_N */


Функция SUB_NDN_N (A, B, k)
	sub:= MUL_ND_N(B,k) // B * k
	// Проверка, что A действительно больше или равно B*k
	Если (COM_NN_D(A, sub) != 1) 
		sub:= SUB_NN_N(A, sub)
		возврат sub
	Иначе
		Ошибка
