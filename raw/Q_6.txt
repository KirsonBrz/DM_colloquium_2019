ФУНКЦИЯ SUB_QQ_Q (q1, q2)
	LCM := LCM_NN_N (q1.n, q2.n) // НОК знаменателей

	t := DIV_NN_N (NOK, q1.n) // деление НОК'а на знаменатель 1-ой дроби
	t := TRANS_N_Z (t) // перевод из натурального в целое 
	p1 := MUL_ZZ_Z (q1.m, t) // домножение числителя p1 на частное НОК'а и знаменателя 1-ой дроби t

	t := DIV_NN_N (NOK, q2.n) // деление НОК'а на знаменатель 2-ой дроби
	t := TRANS_N_Z (t) // перевод из натурального в целое 
	p2 := MUL_ZZ_Z (q2.m, t) // домножение числителя p2 на частное НОК'а и знаменателя 2-ой дроби t

	p3 := SUB_ZZ_Z (p1, p2) // сложение числителей
	
	// Запись значений в новую дробь
	создание дроби q3
	q3.m := p3 
	q3.n := LCM

	RED_Q_Q(q3) // сокращение дроби

	ВОЗВРАТ q3
